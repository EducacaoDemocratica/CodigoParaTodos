Algoritmo "Q8 - exCondicional"

Var
n1, n2, d: real

Inicio
    escreva("Insira um número: ")
    leia(n1)
    escreva("Insira um número: ")
    leia(n2)
    escreval

    se (n2 = 0) entao
        escreval("A divisão de n1 por n2 é indefinida pois n2 = 0.")
    senao
        d <- n1/n2
        escreval("O resultado da divisão de", n1," por", n2," é", d,".")
    fimse
