Algoritmo "Q11 - Sequencial"
Var
   n, n1, n2, n3, n4, n5, mh: real
Inicio
   n <- 5
   escreva("Insira o 1º número: ")
   leia(n1)
   escreva("Insira o 2º número: ")
   leia(n2)
   escreva("Insira o 3º número: ")
   leia(n3)
   escreva("Insira o 4º número: ")
   leia(n4)
   escreva("Insira o 5º número: ")
   leia(n5)
   escreval
   mh <- n / ((1/n1) + (1/n2) + (1/n3) + (1/n4) + (1/n5))
   escreval("A média harmônica do conjunto é", mh, ".")
Fimalgoritmo
