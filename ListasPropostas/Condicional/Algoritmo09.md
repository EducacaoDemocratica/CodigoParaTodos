Algoritmo "Q9 - Condicional"
Var
   n: real
Inicio
   escreva("Insira um número: ")
   leia(n)
   escreval

   se (((n >= -2) e (n <= 18)) ou ((n >= -6) e (n < 9))) entao
      escreval("O número se encontra no intervalo [-2, 18] U [-6, 9[.")
   senao
      escreval("O número não se encontra no intervalo [-2, 18] U [-6, 9[.")
   fimse
