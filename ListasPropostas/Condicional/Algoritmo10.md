Algoritmo "Q10 - Condicional"
Var
   n: real
Inicio
   escreva("Insira um número: ")
   leia(n)
   escreval

   se (((n > -2) e (n <= 18)) e ((n >= -6) e (n < 9))) entao
      escreval("O número se encontra no intervalo [-2, 18] interseção [-6, 9[.")
   senao
      escreval("O número não se encontra no intervalo [-2, 18] interseção [-6, 9[.")
   fimse
Fimalgoritmo
